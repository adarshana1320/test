#Average
numlist = list()
try:
    while True:
        inp = input('Enter a number:')
        if inp == 'done' : break
        value = float(inp)
        numlist.append(value)
    average = sum(numlist)/len(numlist)
    print('Average:', average)
except:
    print('Value could not be found')
